```python
def numOfUnplacedFruits(self, fruits: List[int], baskets: List[int]) -> int:
	remaining = 0
	for fruit in fruits:
		placed = False
		for i in range(len(baskets)):
		if fruit <= baskets[i]:
			baskets[i] = 0
			placed = True
			break
		if not placed:
			remaining += 1
	
	return remaining
```
